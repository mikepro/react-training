  •

let myAction = (parm1, parm2) => ({type:"actionName",parm1, parm2});

Reducer

						
function reducer(state =initialState, action) {
	if (action.type==="myActionName") {
		// do not mutate original state
		// return update state
		return Object.assign({}, state, {key1: action.key1, key2: action.key2});
	}
	return state
	}
